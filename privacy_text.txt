}); } finally { smsContent = null; // ✅ finally 블록으로 이중 보장 } } Vercel 로그 설정 — 원문 로그 차단 (vercel.json) { &quot;functions&quot;: { &quot;api/analyze.ts&quot;: { &quot;maxDuration&quot;: 30 } }, &quot;headers&quot;: [ { &quot;source&quot;: &quot;/api/(.*)&quot;, &quot;headers&quot;: [ { &quot;key&quot;: &quot;X-Content-Type-Options&quot;, &quot;value&quot;: &quot;nosniff&quot;
